#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    std::vector<int> v = {5, 2, 8, 3, 1};
    
    // Sort and print
    std::sort(v.begin(), v.end());
    for (int n : v) std::cout << n << " ";
    
    // Find first even
    auto it = std::find_if(v.begin(), v.end(),
        [](int n){ return n%2 == 0; });
    
    if (it != v.end())
        std::cout << "\nEven: " << *it;
    
    return 0;
}
